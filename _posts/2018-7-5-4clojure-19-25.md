  (if (next s)
          (recur (next s))
          (first s)))
  
(deftest test-19
  (is (= (my-last [1 2 3 4 5]) 5))
  (is (= (my-last '(5 4 3)) 3))
  (is (= (my-last ["b" "c" "d"]) "d")))
